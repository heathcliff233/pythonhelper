lst = [1,2,4,3,5]
for x in lst :

        lst.remove(x)

# 问题：指针删除完元素下移，然而删除一个对象后下一个元素顶上，故出问题
lst = [1,2,4,3,5]
for x in lst[:] :
    if x % 2 == 0:
        lst.remove(x)
print(lst)
get [1,3,5]
